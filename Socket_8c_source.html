Sensor Data Handler System 1.0
This file implements the main entry point and configuration setup for the Kongsberg Maritime's sensor data handling system. The system is designed to collect and store sensor data with flexible and modular configuration capabilities.\n \n This system can be configured to handle multiple data handlers, each with their own thread group. The system is designed to be easily extensible and configurable through JSON configuration files.\n \n Changes to the code can be made to extend the system's capabilities, such as adding protocols. This system has built up a framework for adding new configurations later.

106SocketRecvWithTimeout(int SockFd, void *Buffer, size_t Length, sdb_timediff TimeoutDiff)
107{
108 fd_set ReadFds;
109 struct timeval Timeout;
110
111 // Setup for select()
112 FD_ZERO(&ReadFds);
113 FD_SET(SockFd, &ReadFds);
114
115 SdbTimeval(&Timeout, TimeoutDiff);
116 // Wait for data or timeout
117 int SelectResult = select(SockFd + 1, &ReadFds, NULL, NULL, &Timeout);
118
119 if(SelectResult == -1) {
120 // Select error
121 SdbLogError("Select failed: %s", strerror(errno));
122 return -1;
123 } else if(SelectResult == 0) {
124 // Timeout
125 SdbLogError("Receive timeout after %lu milliseconds", SDB_TIME_TO_MS(TimeoutDiff));
126 return -2;
127 }
128
129 // Data is available, perform the recv
130 int RecvResult = recv(SockFd, Buffer, Length, 0);
131
132 if(RecvResult == 0) {
133 // Server closed connection
134 SdbLogWarning("Server closed connection");
135 return RecvResult;
136 } else if(RecvResult == -1) {
137 SdbLogError("Recv failed: %s", strerror(errno));
138 return -1;
139 }
140
141 return RecvResult;
142}
