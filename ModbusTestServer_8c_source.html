Sensor Data Handler System  1.0
This file implements the main entry point and configuration setup for the Kongsberg Maritime's sensor data handling system. The system is designed to collect and store sensor data with flexible and modular configuration capabilities.\n \n This system can be configured to handle multiple data handlers, each with their own thread group. The system is designed to be easily extensible and configurable through JSON configuration files.\n \n Changes to the code can be made to extend the system's capabilities, such as adding protocols. This system has built up a framework for adding new configurations later.

156 
157  // Sanity check the sizes
158  if(Length > MODBUS_TCP_FRAME_MAX_SIZE - MODBUS_TCP_HEADER_LEN) {
159  SdbLogError("Frame too large: %u", Length);

161  }

164  GenerateShaftPowerDataRandom(&SpData);
165 
166  u16 Pos = 0;
167  // Fixed header construction
168  ModbusFrame[Pos++] = 0x00; // Transaction ID high
169  ModbusFrame[Pos++] = 0x01; // Transaction ID low
170  ModbusFrame[Pos++] = 0x00; // Protocol ID high
171  ModbusFrame[Pos++] = 0x00; // Protocol ID low
172  ModbusFrame[Pos++] = (Length >> 8) & 0xFF; // Length high
173  ModbusFrame[Pos++] = Length & 0xFF; // Length low
174  ModbusFrame[Pos++] = 0x01; // Unit ID
175  ModbusFrame[Pos++] = 0x10; // Function code
176  ModbusFrame[Pos++] = DataLength; // Byte count
177 
178  // Copy data with bounds checking
179  if(Pos + DataLength > MODBUS_TCP_FRAME_MAX_SIZE) {
180  SdbLogError("Buffer overflow prevented");
181  return -1;
182  }
183 
184  SdbMemcpy(&ModbusFrame[Pos], &SpData, DataLength);
185  size_t TotalSize = MODBUS_TCP_HEADER_LEN + Length;
186 
187  SdbLogDebug("Sending frame - Length: %u, DataLength: %u, TotalSize: %zu", Length, DataLength,
188  TotalSize);
189 
190  ssize_t SendResult = send(NewFd, ModbusFrame, TotalSize, 0);
191  if(SendResult > 0) {
192  SendCount++;
193  if(SendCount % 10000 == 0) {
194  SdbLogInfo("Sent %lu packets", SendCount);
195  }
196  }
