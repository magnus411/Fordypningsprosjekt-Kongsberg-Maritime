Sensor Data Handler System 1.0
This file implements the main entry point and configuration setup for the Kongsberg Maritime's sensor data handling system. The system is designed to collect and store sensor data with flexible and modular configuration capabilities.\n \n This system can be configured to handle multiple data handlers, each with their own thread group. The system is designed to be easily extensible and configurable through JSON configuration files.\n \n Changes to the code can be made to extend the system's capabilities, such as adding protocols. This system has built up a framework for adding new configurations later.

116 dprintf(DumpFd, "%s\n", BacktraceSymbols[i]);
117 }
118 free(BacktraceSymbols);

121 FILE *Maps = fopen("/proc/self/maps", "r");
122 if(Maps) {

127 unsigned long Start, End;
128 char Perms[5];
129
130 if(sscanf(Line, "%lx-%lx %4s", &Start, &End, Perms) == 3) {
131 if(strchr(Perms, 'r')) {
132 write(DumpFd, "\n--- Region: ", 12);
133 write(DumpFd, Line, strlen(Line));
134
135 if((End - Start) < 1024 * 1024) { // 1MB limit per region
136 if(write(DumpFd, (void *)Start, End - Start) == -1) {
137 dprintf(DumpFd, "Failed to dump region: %s\n", strerror(errno));
138 }
139 } else {
140 dprintf(DumpFd, "Region too large to dump\n");
141 }
142 }
143 }
144 }
145 fclose(Maps);
146 }

211static void
212SignalHandler(int SignalNum, siginfo_t *Info, void *Context)
213{
214 switch(SignalNum) {
215 case SIGINT:
216 case SIGTERM:
217 {
218 SdbLogWarning("Received termination signal %d", SignalNum);
219 InitiateGracefulShutdown();
220 }
221 break;
222 case SIGSEGV:
223 {
224 SdbLogError("Segmentation fault on thread at address %p", Info->si_addr);
225 DumpProcessMemory(SignalNum, Info);
226 HandlePipeDump();
227 raise(SIGSEGV);
228 }
229 break;
230 case SIGABRT:
231 {
232 SdbLogError("Abnormal termination");
233 DumpProcessMemory(SignalNum, Info);
234 raise(SIGABRT);
235 }
236 break;
237 case SIGFPE:
238 {
239 SdbLogError("Floating point exception on code=%d", Info->si_code);
240 DumpProcessMemory(SignalNum, Info);
241 HandlePipeDump();
242 raise(SIGFPE);
243 }
244 break;
245 case SIGILL:
246 {
247 SdbLogError("Illegal instruction at address %p", Info->si_addr);
248 DumpProcessMemory(SignalNum, Info);
249 raise(SIGILL);
250 }
251 break;
252 default:
253 SdbLogWarning("Unhandled signal %d received", SignalNum);
254 break;
255 }
256}

267int
268SdbSetupSignalHandlers(struct tg_manager *Manager)
269{
270 GSignalContext.Manager = Manager;
271
272 struct sigaction SignalAction;
273 memset(&SignalAction, 0, sizeof(SignalAction));
274 SignalAction.sa_sigaction = SignalHandler;
275 SignalAction.sa_flags = SA_SIGINFO | SA_RESTART;
276
277 sigemptyset(&SignalAction.sa_mask);
278
279 int SignalsToHandle[] = { SIGINT, SIGTERM, SIGSEGV, SIGABRT, SIGFPE, SIGILL };
280
281 for(size_t i = 0; i < sizeof(SignalsToHandle) / sizeof(SignalsToHandle[0]); i++) {
282 if(sigaction(SignalsToHandle[i], &SignalAction, NULL) == -1) {
283 SdbLogError("Failed to set up signal handler for %d: %s", SignalsToHandle[i],
284 strerror(errno));
285 return -1;
286 }
287 }
288
289 return 0;
290}

323bool
324SdbDumpSensorDataPipe(sensor_data_pipe *Pipe, const char *Filename)
325{
326 if(!Pipe || !Filename) {
327 SdbLogError("Invalid pipe or filename for dump");
328 return false;
329 }
330
331 FILE *DumpFile = fopen(Filename, "wb");
332 if(!DumpFile) {
333 SdbLogError("Failed to create pipe dump file: %s", strerror(errno));
334 return false;
335 }
336
337 fwrite(&Pipe->BufCount, sizeof(Pipe->BufCount), 1, DumpFile);
338 fwrite(&Pipe->BufferMaxFill, sizeof(Pipe->BufferMaxFill), 1, DumpFile);
339 fwrite(&Pipe->PacketSize, sizeof(Pipe->PacketSize), 1, DumpFile);
340 fwrite(&Pipe->ItemMaxCount, sizeof(Pipe->ItemMaxCount), 1, DumpFile);
341
342 for(u64 i = 0; i < Pipe->BufCount; i++) {
343 sdb_arena *Buffer = Pipe->Buffers[i];
344 fwrite(&Buffer->Cur, sizeof(Buffer->Cur), 1, DumpFile);
345
346 if(Buffer->Cur > 0) {
347 fwrite(Buffer->Mem, 1, Buffer->Cur, DumpFile);
348 }
349 }
350
351 fclose(DumpFile);
352 SdbLogInfo("Complete sensor data pipe dump saved to %s", Filename);
353 return true;
354}
