Sensor Data Handler System 1.0
This file implements the main entry point and configuration setup for the Kongsberg Maritime's sensor data handling system. The system is designed to collect and store sensor data with flexible and modular configuration capabilities.\n \n This system can be configured to handle multiple data handlers, each with their own thread group. The system is designed to be easily extensible and configurable through JSON configuration files.\n \n Changes to the code can be made to extend the system's capabilities, such as adding protocols. This system has built up a framework for adding new configurations later.

75 // Check if the table exists
76 const char *TableCheckQuery

88 PQclear(TableCheckResult);

141pg_col_metadata *
142GetTableMetadata(PGconn *DbConn, sdb_string TableName, i16 *ColCount, i16 *ColCountNoAutoIncrements,
143 size_t *RowSize, sdb_arena *A)
144{
145 sdb_arena *Conflicts[1] = { A };
146 sdb_scratch_arena Scratch = SdbScratchGet(Conflicts, 1);
147
148 sdb_string MetadataQuery = SdbStringMake(Scratch.Arena, NULL);
149 SdbStringAppendFmt(MetadataQuery, PQ_TABLE_METADATA_QUERY_FMT, TableName);
150
151 PGresult *Result = PQexec(DbConn, MetadataQuery);
152 if(PQresultStatus(Result) != PGRES_TUPLES_OK) {
153 SdbLogError("Metadata query failed: %s", PQerrorMessage(DbConn));
154 PQclear(Result);
155 SdbScratchRelease(Scratch);
156 return NULL;
157 }
158 SdbScratchRelease(Scratch);
159
160 int RowCount = PQntuples(Result);
161 // NOTE(ingar): Each row in the result is the metadata for one column in the requested table
162 *ColCount = RowCount;
163
164 pg_col_metadata *MetadataArray = SdbPushArray(A, pg_col_metadata, RowCount);
165 if(!MetadataArray) {
166 SdbLogError("Memory allocation for PG metadata failed");
167 PQclear(Result);
168 return NULL;
169 }
170
171
172 *ColCountNoAutoIncrements = 0;
173 i32 Offset = 0;
174 for(u64 RowIndex = 0; RowIndex < RowCount; ++RowIndex) {
175 pg_col_metadata *CurrentMetadata = &MetadataArray[RowIndex];
176
177 CurrentMetadata->ColumnName = SdbStringMake(A, PQgetvalue(Result, RowIndex, 0));
178 CurrentMetadata->TypeOid = (pg_oid)atoi(PQgetvalue(Result, RowIndex, 1));
179 CurrentMetadata->TypeLength = (i32)atoi(PQgetvalue(Result, RowIndex, 2));
180 CurrentMetadata->TypeModifier = (i32)atoi(PQgetvalue(Result, RowIndex, 3));
181 CurrentMetadata->IsPrimaryKey = (PQgetvalue(Result, RowIndex, 4)[0] == 't');
182 CurrentMetadata->IsAutoIncrement = (PQgetvalue(Result, RowIndex, 5)[0] == 't');
183
184 if(CurrentMetadata->IsAutoIncrement) {
185 // NOTE(ingar): The assumption is that auto-incrementing values will not appear in the
186 // incoming data
187 CurrentMetadata->Offset = -1;
188 continue;
189 }
190
191 CurrentMetadata->Offset = Offset;
192 Offset += CurrentMetadata->TypeLength;
193 *RowSize += CurrentMetadata->TypeLength;
194 *ColCountNoAutoIncrements += 1;
195 }
196
197 PQclear(Result);
198
199 return MetadataArray;
200}

204PgPrepareCtx(sdb_arena *PgArena, sensor_data_pipe *Pipe)
205{

210 sdb_file_data *ConfFile = SdbLoadFileIntoMemory(POSTGRES_CONF_FS_PATH, Scratch.Arena);
211 if(ConfFile == NULL) {
212 SdbLogError("Failed to open config file");
213 SdbScratchRelease(Scratch);
214 return NULL;
215 }
216
217 cJSON *SchemaConf = DbInitGetConfFromFile("./configs/sensor_schemas.json", Scratch.Arena);
218 if(!cJSON_IsObject(SchemaConf)) {
219 Errno = -SDBE_JSON_ERR;
220 goto cleanup;
221 }
222
223 cJSON *SensorSchemaArray = cJSON_GetObjectItem(SchemaConf, "sensors");
224 if(SensorSchemaArray == NULL || !cJSON_IsArray(SensorSchemaArray)) {
225 Errno = -SDBE_JSON_ERR;
226 goto cleanup;
227 }
228
229 u64 SensorCount = cJSON_GetArraySize(SensorSchemaArray);
230 postgres_ctx *PgCtx = SdbPushStruct(PgArena, postgres_ctx);
231 PgCtx->TablesInfo = SdbPushArray(PgArena, pg_table_info *, SensorCount);
232
233 const char *ConnInfo = (const char *)ConfFile->Data;
234 PgCtx->DbConn = PQconnectdb(ConnInfo);
235
236 if(PQstatus(PgCtx->DbConn) != CONNECTION_OK) {
237 SdbLogError("PgCtx->DbConnection to database failed. PQ error:\n%s",
238 PQerrorMessage(PgCtx->DbConn));
239 Errno = -SDBE_PG_ERR;
240 goto cleanup;
241 }
242
243 u64 SensorIdx = 0;
244 cJSON *SensorSchema = NULL;
245 cJSON_ArrayForEach(SensorSchema, SensorSchemaArray)
246 {
247 if(!cJSON_IsObject(SensorSchema)) {
248 Errno = -SDBE_JSON_ERR;
249 goto cleanup;
250 }
251
252 cJSON *SensorName = cJSON_GetObjectItem(SensorSchema, "name");
253 if(SensorName == NULL || !cJSON_IsString(SensorName)) {
254 Errno = -SDBE_JSON_ERR;
255 goto cleanup;
256 }
257
258
259 pg_table_info *Ti = SdbPushStruct(PgArena, pg_table_info);
260 Ti->TableName = SdbStringMake(PgArena, SensorName->valuestring);
261 PgCtx->TablesInfo[SensorIdx] = Ti;
262
263 sdb_string CreationQuery = SdbStringMake(Scratch.Arena, "CREATE TABLE IF NOT EXISTS ");
264 SdbStringAppendC(CreationQuery, SensorName->valuestring);
265 SdbStringAppendC(CreationQuery, "(\nid SERIAL PRIMARY KEY,\n");
266
267 cJSON *SensorData = cJSON_GetObjectItem(SensorSchema, "data");
268 if(SensorData == NULL || !cJSON_IsObject(SensorData)) {
269 Errno = -SDBE_JSON_ERR;
270 goto cleanup;
271 }
272
273 cJSON *DataAttribute = NULL;
274 cJSON_ArrayForEach(DataAttribute, SensorData)
275 {
276 if(!cJSON_IsString(DataAttribute)) {
277 Errno = -SDBE_JSON_ERR;
278 goto cleanup;
279 }
280
281 SdbStringAppendC(CreationQuery, DataAttribute->string);
282 SdbStringAppendC(CreationQuery, " ");
283 SdbStringAppendC(CreationQuery, DataAttribute->valuestring);
284 SdbStringAppendC(CreationQuery, ",\n");
285 }
286
287 SdbStringBackspace(CreationQuery, 2);
288 SdbStringAppendC(CreationQuery, ");");
289 SdbPrintfDebug("Table creation query:\n%s\n", CreationQuery);
290
291 PGresult *PgRes = PQexec(PgCtx->DbConn, CreationQuery);
292 if(PQresultStatus(PgRes) != PGRES_COMMAND_OK) {
293 SdbLogError("Table creation failed: %s", PQerrorMessage(PgCtx->DbConn));
294 Errno = -SDBE_PG_ERR;
295 PQclear(PgRes);
296 goto cleanup;
297 } else {
298 SdbLogInfo("Table '%s' created successfully (or it already existed).",
299 SensorName->valuestring);
300 PQclear(PgRes);
301 }
302
303
304 Ti->ColMetadata = GetTableMetadata(PgCtx->DbConn, Ti->TableName, &Ti->ColCount,
305 &Ti->ColCountNoAutoIncrements, &Ti->RowSize, PgArena);
306
307 if(NULL == Ti->ColMetadata || 0 == Ti->ColCount) {
308 SdbLogError("Failed to get column metadata for table %s", Ti->TableName);
309 Errno = -SDBE_PG_ERR;
310 goto cleanup;
311 }
312
313 Ti->CopyCommand = SdbStringMake(PgArena, "COPY ");
314 SdbStringAppend(Ti->CopyCommand, Ti->TableName);
315 SdbStringAppendC(Ti->CopyCommand, "(");
316 for(u64 c = 0; c < Ti->ColCount; ++c) {
317 pg_col_metadata ColMd = Ti->ColMetadata[c];
318 if(!ColMd.IsAutoIncrement) {
319 SdbStringAppend(Ti->CopyCommand, ColMd.ColumnName);
320 SdbStringAppendC(Ti->CopyCommand, ", ");
321 }
322 }
323 SdbStringBackspace(Ti->CopyCommand, 2);
324 SdbStringAppendC(Ti->CopyCommand, ") FROM STDIN WITH (FORMAT binary)");
325
326 // NOTE(ingar): An assumption made is that each sensor will have its own pipe since we don't
327 // have a method of differentiating packets at the moment, but unfortunately we probably
328 // don't have time to complete this part. This means that the max number of sensors
329 // supported in this current implementation is 1, BUT extending it to support more should be
330 // relatively simple.
331 Pipe->PacketSize = Ti->RowSize;
332 Pipe->ItemMaxCount = Pipe->Buffers[0]->Cap / Pipe->PacketSize;
333 Pipe->BufferMaxFill = Pipe->PacketSize * Pipe->ItemMaxCount;
334
335 ++SensorIdx;
336 }
337
338cleanup:
339 if(Errno == -SDBE_JSON_ERR) {
340 SdbLogError("Encountered JSON error from sensor schema, cJSON error: %s",
341 cJSON_GetErrorPtr());
342 }
343 if(SchemaConf != NULL) {
344 cJSON_Delete(SchemaConf);
345 }
346 if(Errno != 0) {
347 PQfinish(PgCtx->DbConn);
348 }
349
350 SdbScratchRelease(Scratch);
351 return (Errno == 0) ? PgCtx : NULL;
352}

420sdb_errno
421PgInsertData(PGconn *Conn, pg_table_info *Ti, const char *Data, u64 ItemCount)
422{
423 PGresult *PgRes;
424 sdb_errno Ret = 0;
425
426 PgRes = PQexec(Conn, "BEGIN");
427 if(PQresultStatus(PgRes) != PGRES_COMMAND_OK) {
428 SdbLogError("Failed to begin transaction for insertion into table %s. Pg error: %s",
429 Ti->TableName, PQerrorMessage(Conn));
430 PQclear(PgRes);
431 return -SDBE_PG_ERR;
432 }
433 PQclear(PgRes);
434
435
436 PgRes = PQexec(Conn, Ti->CopyCommand);
437 if(PQresultStatus(PgRes) != PGRES_COPY_IN) {
438 SdbLogError("Failed to start COPY operation for table %s. Pg error: %s", Ti->TableName,
439 PQerrorMessage(Conn));
440 PQclear(PgRes);
441 PQexec(Conn, "ROLLBACK");
442 return -SDBE_PG_ERR;
443 }
444 PQclear(PgRes);
445
446
447 char CopyHeader[19] = "PGCOPY\n\377\r\n\0";
448 u32 CopyFlags = htonl(0);
449 u32 CopyExtenstionLength = htonl(0);
450
451 SdbMemcpy(CopyHeader + 11, &CopyFlags, sizeof(CopyFlags));
452 SdbMemcpy(CopyHeader + 15, &CopyExtenstionLength, sizeof(CopyExtenstionLength));
453
454 if(PQputCopyData(Conn, CopyHeader, 19) != 1) {
455 SdbLogError("Failed to send COPY binary header for table %s. Pg error: %s", Ti->TableName,
456 PQerrorMessage(Conn));
457 Ret = -SDBE_PG_ERR;
458 goto cleanup;
459 }
460
461 size_t PgHeaderSize
462 = sizeof(Ti->ColCount) + (ItemCount * sizeof(Ti->ColMetadata[0].TypeLength));
463 size_t NtwrkConvBufSize = PgHeaderSize + (ItemCount * Ti->RowSize);
464
465 sdb_scratch_arena NtwrkBufArena = SdbScratchGet(NULL, 0);
466 char *NtwrkConvBuf = SdbPushArrayZero(NtwrkBufArena.Arena, char, NtwrkConvBufSize);
467 if(NtwrkConvBuf == NULL) {
468 SdbLogError("Scratch arena has insufficient space for network conversion buffer (need "
469 "%zd bytes). Re-evaluate arena buffer size",
470 NtwrkConvBufSize);
471 SdbScratchRelease(NtwrkBufArena);
472 goto cleanup;
473 }
474
475 size_t ConvOffset = 0;
476 for(u64 RowsProcessed = 0; RowsProcessed < ItemCount; ++RowsProcessed) {
477 const char *CurrRow = (const char *)Data + (RowsProcessed * Ti->RowSize);
478
479 i16 NtwrkFieldCount = htons(Ti->ColCountNoAutoIncrements);
480 SdbMemcpy(NtwrkConvBuf + ConvOffset, &NtwrkFieldCount, sizeof(NtwrkFieldCount));
481 ConvOffset += sizeof(NtwrkFieldCount);
482
483 for(u64 c = 0; c < Ti->ColCount; ++c) {
484 pg_col_metadata ColMd = Ti->ColMetadata[c];
485
486 if(ColMd.IsAutoIncrement) {
487 continue;
488 }
489
490 i32 NtwrkTypeLen = htobe32(ColMd.TypeLength);
491 SdbMemcpy(NtwrkConvBuf + ConvOffset, &NtwrkTypeLen, sizeof(NtwrkTypeLen));
492 ConvOffset += sizeof(NtwrkTypeLen);
493
494 // WARN: The type length might not necessarily be the same as the type written to the
495 // network conversion buffer, so we might need to add handling of that
496 const char *ColData = CurrRow + ColMd.Offset;
497 switch(ColMd.TypeOid) {
498 case PG_INT8:
499 case PG_FLOAT8:
500 ConvOffset += Write8(NtwrkConvBuf + ConvOffset, ColData);
501 break;
502 case PG_INT4:
503 case PG_FLOAT4:
504 ConvOffset += Write4(NtwrkConvBuf + ConvOffset, ColData);
505 break;
506 case PG_INT2:
507 ConvOffset += Write2(NtwrkConvBuf + ConvOffset, ColData);
508 break;
509 case PG_TIMESTAMP:
510 // NOTE(ingar): Support for incoming timestamps of different types
511 // (timeval, timespec, time_t, other) is probably outside the scope of
512 // this project, since I think it will be quite tricky.
513 // I think it's best if we just set it to be time_t, since it's a single
514 // value (unlike timeval and timespec)
515 ConvOffset += WriteTimestamp(NtwrkConvBuf + ConvOffset, ColData);
516 break;
517 default:
518 SdbLogWarning("Encountered unhandled or invalid PG oid %d", ColMd.TypeOid);
519 break;
520 }
521 }
522 }
523
524 if(PQputCopyData(Conn, NtwrkConvBuf, ConvOffset) != 1) {
525 SdbLogError("Unable to copy converted data for table %s. Pg error: %s", Ti->TableName,
526 PQerrorMessage(Conn));
527
528 Ret = -SDBE_PG_ERR;
529 SdbScratchRelease(NtwrkBufArena);
530 goto cleanup;
531 }
532
533 SdbScratchRelease(NtwrkBufArena);
534
535
536cleanup:
537 if(PQputCopyEnd(Conn, (Ret == 0) ? NULL : "Error during copy") != 1) {
538 SdbLogError("Failed to end COPY for table %s. Pg error: %s", Ti->TableName,
539 PQerrorMessage(Conn));
540 Ret = -SDBE_PG_ERR;
541 }
542
543 PgRes = PQgetResult(Conn);
544 if(PQresultStatus(PgRes) != PGRES_COMMAND_OK) {
545 SdbLogError("COPY command failed for table %s. Pg error: %s", Ti->TableName,
546 PQerrorMessage(Conn));
547 Ret = -SDBE_PG_ERR;
548 }
549 PQclear(PgRes);
550
551 if(Ret == 0) {
552 PgRes = PQexec(Conn, "COMMIT");
553 } else {
554 PgRes = PQexec(Conn, "ROLLBACK");
555 }
556
557 if(PQresultStatus(PgRes) != PGRES_COMMAND_OK) {
558 SdbLogError("Failed to end transaction for table %s. Pg error: %s", Ti->TableName,
559 PQerrorMessage(Conn));
560 Ret = -SDBE_PG_ERR;
561 }
562 PQclear(PgRes);
563
564 if(Ret == 0) {
565 SdbLogDebug("Committed copy transaction for table %s", Ti->TableName);
566 } else {
567 SdbLogWarning("Rolled back copy transaction for table %s", Ti->TableName);
568 }
569
570 return Ret;
571}
